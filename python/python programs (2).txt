1.#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include <math.h>


int find_nth_term(int n, int a, int b, int c) {
    // Base Cases: S(1)=a, S(2)=b, S(3)=c
    if (n == 1) {
        return a;
    }
    if (n == 2) {
        return b;
    }
    if (n == 3) {
        return c;
    }
    
    return find_nth_term(n - 1, a, b, c) + 
           find_nth_term(n - 2, a, b, c) + 
           find_nth_term(n - 3, a, b, c);
}

int main() {
    int n, a, b, c;
  
    if (scanf("%d %d %d %d", &n, &a, &b, &c) == 4) {
        int ans = find_nth_term(n, a, b, c);
        printf("%d", ans);
    }
    
    return 0;
}


2.#include <stdio.h>
#include <string.h>
#include <stdlib.h>


int main() {
   
    char *s = malloc(1001 * sizeof(char));
    
   
    int frequency[10] = {0};

    if (scanf("%s", s) == 1) {
        int length = strlen(s);

       
        for (int i = 0; i < length; i++) {
            
            if (s[i] >= '0' && s[i] <= '9') {
                
                int digit = s[i] - '0';
                frequency[digit]++;
            }
        }

        
        for (int i = 0; i < 10; i++) {
            printf("%d", frequency[i]);
            if (i < 9) {
                printf(" ");
            }
        }
        printf("\n");
    }

    free(s);
    
    return 0;
}



3.#include <stdio.h>
#include <string.h>
#include <stdlib.h>

int main() {
    char *s;
    s = malloc(1001 * sizeof(char));
    
   
    if (scanf("%[^\n]", s) == 1) {
        
        char *word = strtok(s, " ");
        while (word != NULL) {
            printf("%s\n", word);
            word = strtok(NULL, " ");
        }
    }
    
    free(s);
    return 0;
}


4.#include <stdio.h>
#include <stdlib.h>

int main(void) {
    int n;
    if (scanf("%d", &n) != 1) return 0;          

    if (n <= 0) return 0;

    int *arr = malloc(n * sizeof *arr);          
    if (arr == NULL) {
        fprintf(stderr, "Memory allocation failed\n");
        return 1;
    }

    for (int i = 0; i < n; ++i) {
        if (scanf("%d", &arr[i]) != 1) arr[i] = 0;
    }

    for (int i = n - 1; i >= 0; --i) {
        printf("%d", arr[i]);
        if (i > 0) putchar(' ');
    }
    putchar('\n');

    free(arr);                                  
    return 0;
}
